@app.post("/api/excuse")
def get_excuse(request: ExcuseInput):
    global latest_text, latest_label
    english, translated = generate_excuse(request.scenario, request.urgency, request.language)
    latest_text = english
    latest_label = "Excuse"
    excuse_history.append({
        "text": english,
        "time": datetime.now().strftime("%Y-%m-%d %H:%M")
    })

    with open("latest_excuse.txt", "w", encoding="utf-8") as f:
        f.write(english)

    ranking_file = "excuse_ranking.json"
    if not os.path.exists(ranking_file):
        with open(ranking_file, "w", encoding="utf-8") as f:
            json.dump({}, f)

    with open(ranking_file, "r+", encoding="utf-8") as f:
        try:
            data = json.load(f)
            if isinstance(data, list):  # Legacy format fix
                temp = {}
                for entry in data:
                    text = entry.get("text", "")
                    if text:
                        temp[text] = temp.get(text, 0) + 1
                data = temp
        except:
            data = {}

        data[english] = data.get(english, 0) + 1
        f.seek(0)
        f.truncate()
        json.dump(data, f, indent=2, ensure_ascii=False)

    # Save to calendar
    now = datetime.now()
    entry = {
        "text": english,
        "date": now.strftime("%Y-%m-%d"),
        "time": now.strftime("%I:%M:%S %p")
    }
    calendar_file = "excuse_calendar.json"
    if not os.path.exists(calendar_file):
        with open(calendar_file, "w", encoding="utf-8") as f:
            json.dump([], f)

    with open(calendar_file, "r+", encoding="utf-8") as f:
        try:
            data = json.load(f)
            if not isinstance(data, list):
                data = []
        except:
            data = []

        data.append(entry)
        f.seek(0)
        f.truncate()
        json.dump(data, f, indent=2)


    return {"english": english, "translated": translated}

# Central emergency function
def trigger_emergency_internal(recipient_override=None):
    try:
        with open("latest_excuse.txt", "r", encoding="utf-8") as ef:
            excuse = ef.read().strip()
    except:
        excuse = "No excuse available."

    try:
        with open("latest_apology.txt", "r", encoding="utf-8") as af:
            apology = af.read().strip()
    except:
        apology = "No apology available."

    EMAIL_SENDER = os.getenv("EMAIL_USERNAME")
    EMAIL_PASSWORD = os.getenv("EMAIL_PASSWORD")
    DEFAULT_RECIPIENTS = os.getenv("EMAIL_RECIPIENTS", "").split(",")

    recipients = []

    if isinstance(recipient_override, dict):
        if recipient_override.get("email"):
            recipients.append(recipient_override["email"])

    else:
        recipients = [r.strip() for r in DEFAULT_RECIPIENTS if r.strip()]


    def send_email():
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        unique_id = str(uuid.uuid4())[:8]

        email_recipients = []
        if recipients:
            email_recipients = [r for r in recipients if "@" in r]

        if not email_recipients:
            email_recipients = [EMAIL_SENDER]

        msg = EmailMessage()
        msg["Subject"] = f"🚨 Emergency Alert [{timestamp}] - ID: {unique_id}"
        msg["From"] = EMAIL_SENDER
        msg["To"] = ", ".join(email_recipients)

        msg.set_content(f"""🚨 Emergency Triggered

🆔 Alert ID: {unique_id}
🕒 Time: {timestamp}

📝 Excuse:
{excuse}

🙏 Apology:
{apology}
""")

        screenshot_path = "static/screenshot.png"
        if os.path.exists(screenshot_path):
            with open(screenshot_path, "rb") as f:
                msg.add_attachment(f.read(), maintype='image', subtype='png', filename='proof.png')

        try:
            with smtplib.SMTP_SSL("smtp.gmail.com", 465) as smtp:
                smtp.login(EMAIL_SENDER, EMAIL_PASSWORD)
                smtp.send_message(msg)
            print("✅ Email sent to:", email_recipients)

            email_log = {
                "id": unique_id,
                "timestamp": timestamp,
                "excuse": excuse,
                "apology": apology,
                "recipients": email_recipients
            }
            if not os.path.exists("email_log.json"):
                with open("email_log.json", "w", encoding="utf-8") as f:
                    json.dump([], f)

            with open("email_log.json", "r+", encoding="utf-8") as f:
                data = json.load(f)
                data.append(email_log)
                f.seek(0)
                json.dump(data, f, indent=2, ensure_ascii=False)

        except Exception as e:
            print("❌ Email error:", e)



    def play_alert():
        try:
            pygame.mixer.init()
            pygame.mixer.music.load("static/alert.mp3")
            pygame.mixer.music.set_volume(1.0)
            pygame.mixer.music.play()
        except Exception as e:
            print("❌ Sound error:", e)

    def log_emergency():
        log = {
            "timestamp": str(datetime.now()),
            "excuse": excuse,
            "apology": apology,
            "recipients": recipients
        }
        try:
            if not os.path.exists("emergency_log.json"):
                with open("emergency_log.json", "w", encoding="utf-8") as f:
                    json.dump([], f)

            with open("emergency_log.json", "r+", encoding="utf-8") as f:
                data = json.load(f)
                data.append(log)
                f.seek(0)
                json.dump(data, f, indent=2, ensure_ascii=False)
            print("📝 Emergency logged.")
        except Exception as e:
            print("❌ Logging failed:", e)

    threading.Thread(target=send_email).start()
    threading.Thread(target=play_alert).start()
    threading.Thread(target=log_emergency).start()

@app.post("/api/emergency")
def trigger_emergency(payload: EmergencyInput):
    trigger_emergency_internal(payload.model_dump())
    return {"status": "ok"}

@app.post("/api/schedule")
def schedule_emergency(input: ScheduleInput):
    try:
        dt_str = f"{input.date} {input.time}"
        dt = datetime.strptime(dt_str, "%Y-%m-%d %H:%M")
        scheduler.add_job(trigger_emergency_internal, trigger=DateTrigger(run_date=dt), args=[input.model_dump()])
        print(f"⏰ Scheduled emergency at {dt}")
        return {"message": f"🚨 Emergency scheduled at {dt}"}
    except Exception as e:
        print("❌ Scheduling failed:", e)
        return {"message": "❌ Failed to schedule emergency"}

@app.get("/api/rankings")
def get_rankings():
    try:
        with open("excuse_ranking.json", "r", encoding="utf-8") as f:
            data = json.load(f)
        if isinstance(data, list):
            # Convert legacy list to dict
            temp = {}
            for item in data:
                text = item.get("text", "")
                if text:
                    temp[text] = temp.get(text, 0) + 1
            data = temp
        sorted_rankings = sorted(data.items(), key=lambda x: x[1], reverse=True)
        return [{"text": r[0], "count": r[1]} for r in sorted_rankings]
    except Exception as e:
        print("❌ Failed to read rankings:", e)
        return []

# ──────── MEMORY LOOKUP ENDPOINT ────────
@app.get("/api/memory")
def memory_lookup(q: str):
    """
    q = free‑text scenario query (front‑end sends whatever’s typed so far)
    returns up to 5 matching excuses from memory_db
    """
    q = q.strip().lower()
    results = []
    for kw, excuses in memory_db.items():
        if q in kw:
            results.extend(excuses)
    # de‑duplicate + limit
    clean = list(dict.fromkeys(results))[:5]
    return {"matches": clean}

@app.post("/api/clear-rankings")
def clear_rankings():
    with open("excuse_ranking.json", "w", encoding="utf-8") as f:
        json.dump({}, f)
    return {"message": "Top excuses cleared."}

atexit.register(memory_db.close)
